Configure o Firebase (Igual ao Admin) const firebaseConfig = { authDomain: "perfil-jovem--disc.firebaseapp.com", databaseURL: "https://perfil-jovem--disc-default-rtdb.firebaseio.com", projectId: "perfil-jovem--disc" }; firebase.initializeApp(firebaseConfig); const db = firebase.database(); // 2. Função para Validar a Chave async function entrarNoTeste() { const nome = document.getElementById('nome-aluno').value; const chave = document.getElementById('chave-acesso').value.trim().toUpperCase(); if(!nome || !chave) return alert("Preencha todos os campos!"); // Busca as chaves no Firebase const snapshot = await db.ref('lista_chaves').once('value'); let chaves = snapshot.val() || []; const index = chaves.findIndex(c => c.code === chave); if (index !== -1 && !chaves[index].used) { // CHAVE VÁLIDA: Marca como usada para ninguém mais entrar com ela chaves[index].used = true; chaves[index].user = nome; await db.ref('lista_chaves').set(chaves); alert("Acesso Autorizado!"); // Redireciona para o teste window.location.href = "quiz.html"; } else { alert("Chave inválida ou já utilizada por outra pessoa!"); } }
